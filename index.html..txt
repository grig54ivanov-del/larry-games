<script>
let money = 0;
let clickPower = 1;

let level = 1;

let carLevel = 0;
let houseLevel = 0;

let passiveIncome = 0;

function update() {
    document.getElementById("money").innerText = Math.floor(money);
    document.getElementById("level").innerText = level;

    document.getElementById("carLevel").innerText = carLevel;
    document.getElementById("houseLevel").innerText = houseLevel;

    document.getElementById("carIncome").innerText = carLevel * 2;
    document.getElementById("houseIncome").innerText = houseLevel * 5;
}

function clickMoney() {
    money += clickPower;
    update();
}

function upgradeCharacter() {
    if (money >= 100) {
        money -= 100;
        level++;
        clickPower += 1;
        update();
    }
}

function upgradeCar() {
    if (money >= 50) {
        money -= 50;
        carLevel++;
        updateIncome();
        update();
    }
}

function upgradeHouse() {
    if (money >= 150) {
        money -= 150;
        houseLevel++;
        updateIncome();
        update();
    }
}

function updateIncome() {
    passiveIncome = (carLevel * 2) + (houseLevel * 5);
}

setInterval(() => {
    money += passiveIncome / 10;
    update();
}, 100);

update();
</script>
