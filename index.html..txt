<script>
let money = 0;
let click = 1;

let level = 1;
let car = 0;
let house = 0;

let income = 0;

function update(){
    document.getElementById("money").innerText = Math.floor(money);
    document.getElementById("level").innerText = level;
    document.getElementById("car").innerText = car;
    document.getElementById("house").innerText = house;
}

function clickMoney(){
    money += click;
    update();
}

function upgradeCharacter(){
    if(money >= 100){
        money -= 100;
        level++;
        click++;
        update();
    }
}

function buyCar(){
    if(money >= 50){
        money -= 50;
        car++;
        updateIncome();
        update();
    }
}

function buyHouse(){
    if(money >= 150){
        money -= 150;
        house++;
        updateIncome();
        update();
    }
}

function updateIncome(){
    income = car * 2 + house * 5;
}

setInterval(()=>{
    money += income / 10;
    update();
},100);

update();
</script>
